#include<stdio.h>

int main(void) {
    int arr[9];
    int sum = 0;
    
    for(int i=0; i<9; i++) {
        scanf("%d", &arr[i]);
        sum += arr[i];
    }
    
    for(int i = 0; i < 8; i++) {
        for(int j = 0; j < 8-i; j++) {
            if(arr[j] > arr[j+1]) {
                int temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
    
    for(int i=0; i < 9; i++) {
        for(int j = i+1; j < 9; j++) {
            if(arr[i] + arr[j] == sum - 100) {
                for(int k = 0; k < 9; k++) {
                    if(k == i || k == j) {
                        continue;
                    }
                    printf("%d\n", arr[k]);
                }
                return 0;
            }
        }
    }
    return 0;
}
